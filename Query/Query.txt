);

#######################################################################################   Medical TABLE 


CREATE TABLE Medical (
  Med_id         VARCHAR(10)  PRIMARY KEY,
  start_date     DATE         NOT NULL,
  end_date       DATE         NOT NULL,
  submitted_date DATE         NOT NULL,
  Status         ENUM('pending','Approved','Not approved')
                 NOT NULL DEFAULT 'pending',
  Reason         VARCHAR(60),
  Reg_no         VARCHAR(12)  NOT NULL,
  Approved_by    VARCHAR(10)  NULL,

 
  FOREIGN KEY (Reg_no) REFERENCES Student(Reg_no)
      ON DELETE CASCADE
	  ON UPDATE CASCADE,

  CONSTRAINT check_approvedby CHECK (
    (Status = 'Approved' AND Approved_by IS NOT NULL)
    OR (Status IN ('pending','Not approved') AND Approved_by IS NULL)
  ),

  CONSTRAINT check_submit_within_14days CHECK (
    (Status = 'Approved' AND submitted_date <= DATE_ADD(end_date, INTERVAL 14 DAY))
    OR (Status IN ('pending','Not approved'))
  )
);

CREATE OR REPLACE VIEW whole_batch_CA AS
WITH

Best2_quiz AS (
  SELECT
    Reg_no,
    Sub_code,
    (
      COALESCE(MAX(CASE WHEN Exam_code='Q1' THEN Mark END), 0) +
      COALESCE(MAX(CASE WHEN Exam_code='Q2' THEN Mark END), 0) +
      COALESCE(MAX(CASE WHEN Exam_code='Q3' THEN Mark END), 0)
    )
    -
    LEAST(
      COALESCE(MAX(CASE WHEN Exam_code='Q1' THEN Mark END), 0),
      COALESCE(MAX(CASE WHEN Exam_code='Q2' THEN Mark END), 0),
      COALESCE(MAX(CASE WHEN Exam_code='Q3' THEN Mark END), 0)
    ) AS Best2Quiz
  FROM Marks
  WHERE Exam_code IN ('Q1','Q2','Q3')
  GROUP BY Reg_no, Sub_code
),


core AS (
  SELECT
    Reg_no,
    Sub_code,
    COALESCE(MAX(CASE WHEN Exam_code='AS' THEN Mark END), 0) AS AS_raw,
    COALESCE(MAX(CASE WHEN Exam_code='MT' THEN Mark END), 0) AS MT_raw,
    COALESCE(MAX(CASE WHEN Exam_code='MP' THEN Mark END), 0) AS MP_raw
  FROM Marks
  GROUP BY Reg_no, Sub_code
),


components AS (
  SELECT
    c.Reg_no,
    c.Sub_code,
    
    COALESCE(
      CASE WHEN c.Sub_code IN ('ENG1222','ICT1222')
           THEN c.AS_raw * 0.20
           ELSE c.AS_raw * 0.10
      END, 0
    ) AS AS_contrib,
    COALESCE(c.MT_raw * 0.20, 0) AS MT_contrib,
    COALESCE(c.MP_raw * 0.20, 0) AS MP_contrib,
    
    COALESCE(b.Best2Quiz / 200.0 * 10.0, 0) AS BQ_contrib
  FROM core c
  LEFT JOIN Best2_quiz b
    ON b.Reg_no = c.Reg_no AND b.Sub_code = c.Sub_code
),


subject_ca AS (
  SELECT
    Reg_no,
    Sub_code,
    CASE Sub_code
      WHEN 'ENG1222' THEN AS_contrib + MT_contrib
      WHEN 'ICT1212' THEN BQ_contrib + MT_contrib + AS_contrib
      WHEN 'ICT1222' THEN AS_contrib + MP_contrib            
      WHEN 'ICT1233' THEN BQ_contrib + MP_contrib + AS_contrib
      WHEN 'ICT1242' THEN BQ_contrib + MT_contrib + AS_contrib
      WHEN 'ICT1253' THEN BQ_contrib + MP_contrib + AS_contrib
      WHEN 'TCS1212' THEN BQ_contrib + AS_contrib + MT_contrib
      WHEN 'TCS1233' THEN BQ_contrib + MT_contrib + AS_contrib
      ELSE NULL
    END AS ca_raw,
    40 AS ca_max
  FROM components
),


scaled AS (
  SELECT
    Reg_no,
    Sub_code,
    ROUND(
      CASE
        WHEN ca_raw IS NULL THEN 0
        ELSE ca_raw * 40.0 / ca_max   
      END, 2
    ) AS ca_40
  FROM subject_ca
)


SELECT
  s.Reg_no,
  COALESCE(MAX(CASE WHEN Sub_code='ENG1222' THEN ca_40 END), 0) AS ENG1222,
  COALESCE(MAX(CASE WHEN Sub_code='ICT1212' THEN ca_40 END), 0) AS ICT1212,
  COALESCE(MAX(CASE WHEN Sub_code='ICT1222' THEN ca_40 END), 0) AS ICT1222,
  COALESCE(MAX(CASE WHEN Sub_code='ICT1233' THEN ca_40 END), 0) AS ICT1233,
  COALESCE(MAX(CASE WHEN Sub_code='ICT1242' THEN ca_40 END), 0) AS ICT1242,
  COALESCE(MAX(CASE WHEN Sub_code='ICT1253' THEN ca_40 END), 0) AS ICT1253,
  COALESCE(MAX(CASE WHEN Sub_code='TCS1212' THEN ca_40 END), 0) AS TCS1212,
  COALESCE(MAX(CASE WHEN Sub_code='TCS1233' THEN ca_40 END), 0) AS TCS1233
FROM Student s
LEFT JOIN scaled x
  ON x.Reg_no = s.Reg_no WHERE s.Dep_id='ICT' AND s.Batch='2022/2023'
GROUP BY s.Reg_no
ORDER BY s.Reg_no;

###########################################################################################VIEW ATTANDENCE SUMMARY



CREATE OR REPLACE VIEW View_Attendance_Summary AS
WITH Approved_Medical AS (
    SELECT Reg_no, start_date, end_date
    FROM Medical
    WHERE Status = 'Approved'
),
Subject_Total_Hours AS (
    SELECT
        Sub_code,
        COALESCE(SUM(Hour),0) AS Total_Lessons_Hour
    FROM Session
    GROUP BY Sub_code
),
Session_With_Attendance AS (
    SELECT
        ss.Reg_no,
        s.Sub_code,
        s.Session_id,
        s.Hour,
        s.Ses_date,
        ss.Status,
        CASE
            WHEN am.Reg_no IS NOT NULL
                 AND ss.Status = 'Absent'
                 AND s.Ses_date BETWEEN am.start_date AND am.end_date
            THEN 1 ELSE 0
        END AS Is_Medical
    FROM Session_Student ss
    JOIN Session s
      ON ss.Sub_code = s.Sub_code
     AND ss.Session_id = s.Session_id
    LEFT JOIN Approved_Medical am
      ON ss.Reg_no = am.Reg_no
     AND s.Ses_date BETWEEN am.start_date AND am.end_date
),
Student_Summary AS (
    SELECT
        swa.Reg_no,
        swa.Sub_code,
        SUM(CASE WHEN swa.Status = 'Present' THEN swa.Hour ELSE 0 END) AS Participate_Hour,
        SUM(CASE WHEN swa.Is_Medical = 1 AND swa.Status = 'Absent' THEN swa.Hour ELSE 0 END) AS Total_Medical_Submitted_Hour
    FROM Session_With_Attendance swa
    GROUP BY swa.Reg_no, swa.Sub_code
)
SELECT
    ss.Reg_no,
    st.Sub_code,
    subj.Sub_name,
    st.Total_Lessons_Hour,
    COALESCE(ss.Participate_Hour, 0) AS Participate_Hour,
    COALESCE(ss.Total_Medical_Submitted_Hour, 0) AS Total_Medical_Submitted_Hour,
    LEAST(COALESCE(ss.Total_Medical_Submitted_Hour,0), ROUND(0.2 * st.Total_Lessons_Hour, 2)) AS Accepted_Medical_Hour,
    GREATEST(COALESCE(ss.Total_Medical_Submitted_Hour,0) - LEAST(COALESCE(ss.Total_Medical_Submitted_Hour,0), ROUND(0.2 * st.Total_Lessons_Hour, 2)), 0) AS Ignored_Medical_Hour,
    ROUND(
      (LEAST(COALESCE(ss.Total_Medical_Submitted_Hour,0), 0.2 * st.Total_Lessons_Hour) / NULLIF(st.Total_Lessons_Hour,0)) * 100
    , 2) AS Medical_Percentage,
    ROUND(
      ( (COALESCE(ss.Participate_Hour,0) + LEAST(COALESCE(ss.Total_Medical_Submitted_Hour,0), 0.2 * st.Total_Lessons_Hour))
        / NULLIF(st.Total_Lessons_Hour,0)
      ) * 100
    , 2) AS Total_Attendance_Percentage
FROM Subject_Total_Hours st
JOIN Subject subj ON st.Sub_code = subj.Sub_code
LEFT JOIN Student_Summary ss ON st.Sub_code = ss.Sub_code
ORDER BY ss.Reg_no, st.Sub_code;

DELIMITER $$

CREATE PROCEDURE Student_subject_breakdown(
  IN p_reg_no   VARCHAR(12),
  IN p_sub_code VARCHAR(10)
)
BEGIN
  SELECT
    c.Exam_code AS Exam_type,
    CASE
      
      WHEN m.raw_mark IS NULL AND EXISTS (
        SELECT 1 FROM Medical
        WHERE Reg_no = p_reg_no AND Status = 'Approved'
      ) THEN 'MC'

      
      ELSE CAST(ROUND(
        CASE c.Exam_code
          WHEN 'AS' THEN 
            COALESCE(m.raw_mark,0) *
            (CASE WHEN p_sub_code IN ('ENG1222','ICT1222') THEN 0.20 ELSE 0.10 END)
          WHEN 'MT' THEN COALESCE(m.raw_mark,0) * 0.20
          WHEN 'MP' THEN COALESCE(m.raw_mark,0) * 0.20
          WHEN 'Q1' THEN COALESCE(m.raw_mark,0) / 100.0 * 5.0
          WHEN 'Q2' THEN COALESCE(m.raw_mark,0) / 100.0 * 5.0
          WHEN 'Q3' THEN COALESCE(m.raw_mark,0) / 100.0 * 5.0
          ELSE 0
        END, 2) AS CHAR)
    END AS mark
  FROM (
    SELECT 'Q1' AS Exam_code, 1 AS ord UNION ALL
    SELECT 'Q2', 2 UNION ALL
    SELECT 'Q3', 3 UNION ALL
    SELECT 'AS', 4 UNION ALL
    SELECT 'MT', 5 UNION ALL
    SELECT 'MP', 6
  ) AS c

  LEFT JOIN (
    SELECT Exam_code, MAX(Mark) AS raw_mark
    FROM Marks
    WHERE Reg_no = p_reg_no
      AND Sub_code = p_sub_code
      AND Exam_code IN ('Q1','Q2','Q3','AS','MT','MP')
    GROUP BY Exam_code
  ) AS m
    ON c.Exam_code = m.Exam_code

  
  WHERE
    (p_sub_code='ENG1222'  AND c.Exam_code IN ('AS','MT')) OR
    (p_sub_code='ICT1212'  AND c.Exam_code IN ('Q1','Q2','Q3','AS','MT','MP')) OR
    (p_sub_code='ICT1222'  AND c.Exam_code IN ('AS','MP')) OR
    (p_sub_code='ICT1233'  AND c.Exam_code IN ('Q1','Q2','Q3','AS','MP')) OR
    (p_sub_code='ICT1242'  AND c.Exam_code IN ('Q1','Q2','Q3','AS','MT')) OR
    (p_sub_code='ICT1253'  AND c.Exam_code IN ('Q1','Q2','Q3','AS','MP')) OR
    (p_sub_code='TCS1212'  AND c.Exam_code IN ('Q1','Q2','Q3','AS','MT')) OR
    (p_sub_code='TCS1233'  AND c.Exam_code IN ('Q1','Q2','Q3','AS','MT'))
  ORDER BY c.ord;
END $$

################################################################################################### FINAL RESULTS


CREATE OR REPLACE VIEW Final_results AS
WITH
ca_unpivot AS (
  SELECT Reg_no, 'ENG1222' AS Sub_code, ENG1222 AS c_mark FROM whole_batch_CA
  UNION ALL SELECT Reg_no, 'ICT1212', ICT1212 FROM whole_batch_CA
  UNION ALL SELECT Reg_no, 'ICT1222', ICT1222 FROM whole_batch_CA
  UNION ALL SELECT Reg_no, 'ICT1233', ICT1233 FROM whole_batch_CA
  UNION ALL SELECT Reg_no, 'ICT1242', ICT1242 FROM whole_batch_CA
  UNION ALL SELECT Reg_no, 'ICT1253', ICT1253 FROM whole_batch_CA
  UNION ALL SELECT Reg_no, 'TCS1212', TCS1212 FROM whole_batch_CA
  UNION ALL SELECT Reg_no, 'TCS1233', TCS1233 FROM whole_batch_CA
),
exam_setup AS (
  SELECT
    Sub_code,
    MAX(CASE WHEN Exam_code='ET' THEN 1 ELSE 0 END) AS has_et,
    MAX(CASE WHEN Exam_code='EP' THEN 1 ELSE 0 END) AS has_ep
  FROM Exam_Subject
  GROUP BY Sub_code
),
end_exam_raw AS (
  SELECT
    Reg_no,
    Sub_code,
    MAX(CASE WHEN Exam_code='ET' THEN Mark END) AS et_mark,
    MAX(CASE WHEN Exam_code='EP' THEN Mark END) AS ep_mark
  FROM Marks
  WHERE Exam_code IN ('ET','EP')
  GROUP BY Reg_no, Sub_code
),
end_exam AS (
  SELECT
    r.Reg_no,
    r.Sub_code,
    CASE
      WHEN COALESCE(s.has_et,0)=1 AND COALESCE(s.has_ep,0)=1 THEN
        COALESCE(r.et_mark,0)/100*30 + COALESCE(r.ep_mark,0)/100*30
      WHEN COALESCE(s.has_et,0)=1 AND COALESCE(s.has_ep,0)=0 THEN
        CASE WHEN r.et_mark IS NULL THEN NULL ELSE r.et_mark/100*60 END
      WHEN COALESCE(s.has_et,0)=0 AND COALESCE(s.has_ep,0)=1 THEN
        CASE WHEN r.ep_mark IS NULL THEN NULL ELSE r.ep_mark/100*60 END
      ELSE
        NULL
    END AS end_exam_60
  FROM end_exam_raw r
  LEFT JOIN exam_setup s ON s.Sub_code = r.Sub_code
),
mc AS (
  SELECT DISTINCT m.Reg_no, e.Sub_code
  FROM Medical m
  JOIN Exam_Subject e
    ON e.Date BETWEEN m.start_date AND m.end_date
   AND e.Exam_code IN ('ET','EP')
  WHERE m.Status='Approved'
),
subject_grades AS (
  SELECT
    c.Reg_no,
    c.Sub_code,
    ROUND(c.c_mark,2) AS c_mark,
    ROUND(COALESCE(e.end_exam_60,0),2) AS end_exam,
    ROUND(c.c_mark + COALESCE(e.end_exam_60,0),2) AS total,
    CASE
      WHEN m.Reg_no IS NOT NULL THEN 'MC'
      WHEN e.end_exam_60 IS NULL THEN 'E*'
      WHEN COALESCE(e.end_exam_60,0) < 12 THEN 'E'
      ELSE CASE
        WHEN (c.c_mark + COALESCE(e.end_exam_60,0)) >= 85 THEN 'A+'
        WHEN (c.c_mark + COALESCE(e.end_exam_60,0)) >= 75 THEN 'A'
        WHEN (c.c_mark + COALESCE(e.end_exam_60,0)) >= 70 THEN 'A-'
        WHEN (c.c_mark + COALESCE(e.end_exam_60,0)) >= 65 THEN 'B+'
        WHEN (c.c_mark + COALESCE(e.end_exam_60,0)) >= 60 THEN 'B'
        WHEN (c.c_mark + COALESCE(e.end_exam_60,0)) >= 55 THEN 'B-'
        WHEN (c.c_mark + COALESCE(e.end_exam_60,0)) >= 50 THEN 'C+'
        WHEN (c.c_mark + COALESCE(e.end_exam_60,0)) >= 45 THEN 'C'
        WHEN (c.c_mark + COALESCE(e.end_exam_60,0)) >= 40 THEN 'C-'
        WHEN (c.c_mark + COALESCE(e.end_exam_60,0)) >= 35 THEN 'D'
        ELSE 'E'
      END
    END AS Grade,
    CASE
      WHEN m.Reg_no IS NOT NULL THEN NULL
      WHEN e.end_exam_60 IS NULL THEN NULL
      ELSE CASE
        WHEN (c.c_mark + COALESCE(e.end_exam_60,0)) >= 85 THEN 4.00
        WHEN (c.c_mark + COALESCE(e.end_exam_60,0)) >= 75 THEN 4.00
        WHEN (c.c_mark + COALESCE(e.end_exam_60,0)) >= 70 THEN 3.70
        WHEN (c.c_mark + COALESCE(e.end_exam_60,0)) >= 65 THEN 3.30
        WHEN (c.c_mark + COALESCE(e.end_exam_60,0)) >= 60 THEN 3.00
        WHEN (c.c_mark + COALESCE(e.end_exam_60,0)) >= 55 THEN 2.70
        WHEN (c.c_mark + COALESCE(e.end_exam_60,0)) >= 50 THEN 2.30
        WHEN (c.c_mark + COALESCE(e.end_exam_60,0)) >= 45 THEN 2.00
        WHEN (c.c_mark + COALESCE(e.end_exam_60,0)) >= 40 THEN 1.70
        WHEN (c.c_mark + COALESCE(e.end_exam_60,0)) >= 35 THEN 1.30
        
        ELSE 0.00
      END
    END AS GP
  FROM ca_unpivot c
  LEFT JOIN end_exam e ON e.Reg_no=c.Reg_no AND e.Sub_code=c.Sub_code
  LEFT JOIN mc m       ON m.Reg_no=c.Reg_no AND m.Sub_code=c.Sub_code
),

credits AS (
  SELECT Sub_code, 1 AS Credits FROM Subject
),
sgpa AS (
  SELECT
    g.Reg_no,
    ROUND(
      SUM(g.GP * c.Credits) / NULLIF(SUM(CASE WHEN g.GP IS NULL THEN 0 ELSE c.Credits END),0)
    , 2) AS SGPA
  FROM subject_grades g
  JOIN credits c ON c.Sub_code=g.Sub_code
  WHERE g.GP IS NOT NULL
  GROUP BY g.Reg_no
),
colors AS (
  SELECT Reg_no, 0.20 AS bonus
  FROM Student_Sports
  WHERE Having_colors='YES'
  GROUP BY Reg_no
)
SELECT
  s.Reg_no,
  MAX(CASE WHEN s.Sub_code='ENG1222' THEN s.Grade END) AS ENG1222,
  MAX(CASE WHEN s.Sub_code='ICT1212' THEN s.Grade END) AS ICT1212,
  MAX(CASE WHEN s.Sub_code='ICT1222' THEN s.Grade END) AS ICT1222,
  MAX(CASE WHEN s.Sub_code='ICT1233' THEN s.Grade END) AS ICT1233,
  MAX(CASE WHEN s.Sub_code='ICT1242' THEN s.Grade END) AS ICT1242,
  MAX(CASE WHEN s.Sub_code='ICT1253' THEN s.Grade END) AS ICT1253,
  MAX(CASE WHEN s.Sub_code='TCS1212' THEN s.Grade END) AS TCS1212,
  MAX(CASE WHEN s.Sub_code='TCS1233' THEN s.Grade END) AS TCS1233,
  sg.SGPA,
  LEAST(sg.SGPA + COALESCE(cz.bonus,0), 4.00) AS CGPA
FROM subject_grades s
JOIN sgpa sg     ON sg.Reg_no = s.Reg_no
LEFT JOIN colors cz ON cz.Reg_no = s.Reg_no
GROUP BY s.Reg_no, sg.SGPA, cz.bonus;

CREATE OR REPLACE VIEW 8th_batch_marks_summary AS
WITH

exam_setup AS (
  SELECT
    Sub_code,
    MAX(CASE WHEN Exam_code='ET' THEN 1 ELSE 0 END) AS has_et,
    MAX(CASE WHEN Exam_code='EP' THEN 1 ELSE 0 END) AS has_ep
  FROM Exam_Subject
  GROUP BY Sub_code
),


end_raw AS (
  SELECT
    Reg_no,
    Sub_code,
    MAX(CASE WHEN Exam_code='ET' THEN Mark END) AS et_mark,
    MAX(CASE WHEN Exam_code='EP' THEN Mark END) AS ep_mark
  FROM Marks
  WHERE Exam_code IN ('ET','EP')
  GROUP BY Reg_no, Sub_code
),


end_60 AS (
  SELECT
    r.Reg_no,
    r.Sub_code,
    CASE
      WHEN COALESCE(s.has_et,0)=1 AND COALESCE(s.has_ep,0)=1 THEN
        COALESCE(r.et_mark,0)/100*30 + COALESCE(r.ep_mark,0)/100*30
      WHEN COALESCE(s.has_et,0)=1 AND COALESCE(s.has_ep,0)=0 THEN
        CASE WHEN r.et_mark IS NULL THEN 0 ELSE r.et_mark/100*60 END
      WHEN COALESCE(s.has_et,0)=0 AND COALESCE(s.has_ep,0)=1 THEN
        CASE WHEN r.ep_mark  IS NULL THEN 0 ELSE r.ep_mark /100*60 END
      ELSE
        0  
    END AS end_contrib_60
  FROM end_raw r
  LEFT JOIN exam_setup s ON s.Sub_code = r.Sub_code

  UNION ALL
  SELECT
    w.Reg_no,
    sc.Sub_code,
    0 AS end_contrib_60
  FROM whole_batch_CA w
  JOIN (
    SELECT 'ENG1222' AS Sub_code UNION ALL
    SELECT 'ICT1212' UNION ALL
    SELECT 'ICT1222' UNION ALL
    SELECT 'ICT1233' UNION ALL
    SELECT 'ICT1242' UNION ALL
    SELECT 'ICT1253' UNION ALL
    SELECT 'TCS1212' UNION ALL
    SELECT 'TCS1233'
  ) sc
  LEFT JOIN end_raw er ON er.Reg_no=w.Reg_no AND er.Sub_code=sc.Sub_code
  WHERE er.Reg_no IS NULL
)

SELECT 
  w.Reg_no,

  ROUND(w.ENG1222 + COALESCE(MAX(CASE WHEN e.Sub_code='ENG1222' THEN e.end_contrib_60 END),0), 2) AS ENG1222,
  ROUND(w.ICT1212 + COALESCE(MAX(CASE WHEN e.Sub_code='ICT1212' THEN e.end_contrib_60 END),0), 2) AS ICT1212,
  ROUND(w.ICT1222 + COALESCE(MAX(CASE WHEN e.Sub_code='ICT1222' THEN e.end_contrib_60 END),0), 2) AS ICT1222,
  ROUND(w.ICT1233 + COALESCE(MAX(CASE WHEN e.Sub_code='ICT1233' THEN e.end_contrib_60 END),0), 2) AS ICT1233,
  ROUND(w.ICT1242 + COALESCE(MAX(CASE WHEN e.Sub_code='ICT1242' THEN e.end_contrib_60 END),0), 2) AS ICT1242,
  ROUND(w.ICT1253 + COALESCE(MAX(CASE WHEN e.Sub_code='ICT1253' THEN e.end_contrib_60 END),0), 2) AS ICT1253,
  ROUND(w.TCS1212 + COALESCE(MAX(CASE WHEN e.Sub_code='TCS1212' THEN e.end_contrib_60 END),0), 2) AS TCS1212,
  ROUND(w.TCS1233 + COALESCE(MAX(CASE WHEN e.Sub_code='TCS1233' THEN e.end_contrib_60 END),0), 2) AS TCS1233

FROM whole_batch_CA w
LEFT JOIN end_60 e ON e.Reg_no = w.Reg_no
GROUP BY w.Reg_no
ORDER BY w.Reg_no;

CREATE PROCEDURE sp_view_subject_marks(IN p_sub_code VARCHAR(10))
BEGIN

  SELECT 
      w.Reg_no,

      COALESCE(
        CASE p_sub_code
          WHEN 'ENG1222' THEN w.ENG1222
          WHEN 'ICT1212' THEN w.ICT1212
          WHEN 'ICT1222' THEN w.ICT1222
          WHEN 'ICT1233' THEN w.ICT1233
          WHEN 'ICT1242' THEN w.ICT1242
          WHEN 'ICT1253' THEN w.ICT1253
          WHEN 'TCS1212' THEN w.TCS1212
          WHEN 'TCS1233' THEN w.TCS1233

      ) AS CA_40,

      LEAST(
        60,
        GREATEST(
          COALESCE(
            CASE p_sub_code
              WHEN 'ENG1222' THEN s.ENG1222
              WHEN 'ICT1212' THEN s.ICT1212
              WHEN 'ICT1222' THEN s.ICT1222
              WHEN 'ICT1233' THEN s.ICT1233
              WHEN 'ICT1242' THEN s.ICT1242
              WHEN 'ICT1253' THEN s.ICT1253
              WHEN 'TCS1212' THEN s.TCS1212
              WHEN 'TCS1233' THEN s.TCS1233
            END, 0
          ) 
          - COALESCE(
            CASE p_sub_code
              WHEN 'ENG1222' THEN w.ENG1222
              WHEN 'ICT1212' THEN w.ICT1212
              WHEN 'ICT1222' THEN w.ICT1222
              WHEN 'ICT1233' THEN w.ICT1233
              WHEN 'ICT1242' THEN w.ICT1242
              WHEN 'ICT1253' THEN w.ICT1253
              WHEN 'TCS1212' THEN w.TCS1212
              WHEN 'TCS1233' THEN w.TCS1233
            END, 0
          ),
          0
        )
      ) AS End_60,

      COALESCE(
        CASE p_sub_code
          WHEN 'ENG1222' THEN s.ENG1222
          WHEN 'ICT1212' THEN s.ICT1212
          WHEN 'ICT1222' THEN s.ICT1222
          WHEN 'ICT1233' THEN s.ICT1233
          WHEN 'ICT1242' THEN s.ICT1242
          WHEN 'ICT1253' THEN s.ICT1253
          WHEN 'TCS1212' THEN s.TCS1212
          WHEN 'TCS1233' THEN s.TCS1233
        END, 0
      ) AS Total_100

  LEFT JOIN v_whole_batch_marks_summary s ON s.Reg_no = w.Reg_no
  ORDER BY w.Reg_no;

CREATE OR REPLACE VIEW Final_results_repeat_only AS
WITH
subject_list AS (
  SELECT 'ENG1222' AS Sub_code UNION ALL
  SELECT 'ICT1212' UNION ALL
  SELECT 'ICT1222' UNION ALL
  SELECT 'ICT1233' UNION ALL
  SELECT 'ICT1242' UNION ALL
  SELECT 'ICT1253' UNION ALL
  SELECT 'TCS1212' UNION ALL
  SELECT 'TCS1233'
),

repeat_flag AS (
  SELECT DISTINCT Reg_no, Sub_code
  FROM Marks
  WHERE Type = 'Repeat'
),
repeat_students AS (
  SELECT DISTINCT Reg_no FROM repeat_flag
),
student_subjects AS (
  SELECT rs.Reg_no, sl.Sub_code
  FROM repeat_students rs
  CROSS JOIN subject_list sl
),

ca_unpivot AS (
  SELECT Reg_no, 'ENG1222' AS Sub_code, ENG1222 AS ca_40 FROM whole_batch_CA
  UNION ALL SELECT Reg_no, 'ICT1212', ICT1212 FROM whole_batch_CA
  UNION ALL SELECT Reg_no, 'ICT1222', ICT1222 FROM whole_batch_CA
  UNION ALL SELECT Reg_no, 'ICT1233', ICT1233 FROM whole_batch_CA
  UNION ALL SELECT Reg_no, 'ICT1242', ICT1242 FROM whole_batch_CA
  UNION ALL SELECT Reg_no, 'ICT1253', ICT1253 FROM whole_batch_CA
  UNION ALL SELECT Reg_no, 'TCS1212', TCS1212 FROM whole_batch_CA
  UNION ALL SELECT Reg_no, 'TCS1233', TCS1233 FROM whole_batch_CA
),


exam_setup AS (
  SELECT
    Sub_code,
    MAX(CASE WHEN Exam_code='ET' THEN 1 ELSE 0 END) AS has_et,
    MAX(CASE WHEN Exam_code='EP' THEN 1 ELSE 0 END) AS has_ep
  FROM Exam_Subject
  GROUP BY Sub_code
),


end_raw AS (
  SELECT
    Reg_no,
    Sub_code,
    MAX(CASE WHEN Exam_code='ET' THEN Mark END) AS et_mark,
    MAX(CASE WHEN Exam_code='EP' THEN Mark END) AS ep_mark
  FROM Marks
  WHERE Exam_code IN ('ET','EP')
  GROUP BY Reg_no, Sub_code
),


joined AS (
  SELECT
    ss.Reg_no,
    ss.Sub_code,
    COALESCE(ca.ca_40, 0) AS ca_40,
    COALESCE(
      CASE
        WHEN COALESCE(es.has_et,0)=1 AND COALESCE(es.has_ep,0)=1
          THEN COALESCE(er.et_mark,0)/100*30 + COALESCE(er.ep_mark,0)/100*30
        WHEN COALESCE(es.has_et,0)=1 AND COALESCE(es.has_ep,0)=0
          THEN COALESCE(er.et_mark,0)/100*60
        WHEN COALESCE(es.has_et,0)=0 AND COALESCE(es.has_ep,0)=1
          THEN COALESCE(er.ep_mark,0)/100*60
        ELSE 0
      END, 0
    ) AS end_60,
    CASE WHEN (er.et_mark IS NOT NULL OR er.ep_mark IS NOT NULL) THEN 1 ELSE 0 END AS sat_end
  FROM student_subjects ss
  LEFT JOIN ca_unpivot ca ON ca.Reg_no=ss.Reg_no AND ca.Sub_code=ss.Sub_code
  LEFT JOIN end_raw   er  ON er.Reg_no=ss.Reg_no AND er.Sub_code=ss.Sub_code
  LEFT JOIN exam_setup es ON es.Sub_code=ss.Sub_code
),


base AS (
  SELECT
    j.Reg_no, j.Sub_code, j.ca_40, j.end_60, j.sat_end,
    CASE
      WHEN j.sat_end = 0 THEN '-'                               
      WHEN j.end_60  < 12 THEN 'E'                            
      ELSE
        CASE
          WHEN (j.ca_40 + j.end_60) >= 85 THEN 'A+'
          WHEN (j.ca_40 + j.end_60) >= 70 THEN 'A'
          WHEN (j.ca_40 + j.end_60) >= 65 THEN 'A-'
          WHEN (j.ca_40 + j.end_60) >= 60 THEN 'B+'
          WHEN (j.ca_40 + j.end_60) >= 55 THEN 'B'
          WHEN (j.ca_40 + j.end_60) >= 50 THEN 'B-'
          WHEN (j.ca_40 + j.end_60) >= 45 THEN 'C+'
          WHEN (j.ca_40 + j.end_60) >= 40 THEN 'C'
          WHEN (j.ca_40 + j.end_60) >= 35 THEN 'C-'
          WHEN (j.ca_40 + j.end_60) >= 30 THEN 'D+'
          WHEN (j.ca_40 + j.end_60) >= 20 THEN 'D'
          ELSE 'E'
        END
    END AS base_grade,
    CASE
      WHEN j.sat_end = 0 THEN NULL
      WHEN j.end_60  < 12 THEN 0.00
      ELSE
        CASE
          WHEN (j.ca_40 + j.end_60) >= 85 THEN 4.00
          WHEN (j.ca_40 + j.end_60) >= 70 THEN 4.00
          WHEN (j.ca_40 + j.end_60) >= 65 THEN 3.70
          WHEN (j.ca_40 + j.end_60) >= 60 THEN 3.30
          WHEN (j.ca_40 + j.end_60) >= 55 THEN 3.00
          WHEN (j.ca_40 + j.end_60) >= 50 THEN 2.70
          WHEN (j.ca_40 + j.end_60) >= 45 THEN 2.30
          WHEN (j.ca_40 + j.end_60) >= 40 THEN 2.00
          WHEN (j.ca_40 + j.end_60) >= 35 THEN 1.70
          WHEN (j.ca_40 + j.end_60) >= 30 THEN 1.30
          WHEN (j.ca_40 + j.end_60) >= 20 THEN 1.00
          ELSE 0.00
        END
    END AS base_gp
  FROM joined j
),

capped AS (
  SELECT
    b.Reg_no,
    b.Sub_code,
    CASE
      WHEN EXISTS (SELECT 1 FROM repeat_flag r
                   WHERE r.Reg_no=b.Reg_no AND r.Sub_code=b.Sub_code)
           AND b.base_grade <> '-'
           AND b.base_grade IN ('A+','A','A-','B+','B','B-','C+')
        THEN 'C'
      ELSE b.base_grade
    END AS Grade,
    CASE
      WHEN EXISTS (SELECT 1 FROM repeat_flag r
                   WHERE r.Reg_no=b.Reg_no AND r.Sub_code=b.Sub_code)
           AND b.base_grade <> '-'
        THEN LEAST(COALESCE(b.base_gp,0), 2.00)
      ELSE b.base_gp
    END AS GP
  FROM base b
),


sgpa AS (
  SELECT Reg_no, ROUND(AVG(GP),2) AS SGPA
  FROM capped
  WHERE GP IS NOT NULL
  GROUP BY Reg_no
)


SELECT
  c.Reg_no,
  MAX(CASE WHEN c.Sub_code='ENG1222' THEN c.Grade END) AS ENG1222,
  MAX(CASE WHEN c.Sub_code='ICT1212' THEN c.Grade END) AS ICT1212,
  MAX(CASE WHEN c.Sub_code='ICT1222' THEN c.Grade END) AS ICT1222,
  MAX(CASE WHEN c.Sub_code='ICT1233' THEN c.Grade END) AS ICT1233,
  MAX(CASE WHEN c.Sub_code='ICT1242' THEN c.Grade END) AS ICT1242,
  MAX(CASE WHEN c.Sub_code='ICT1253' THEN c.Grade END) AS ICT1253,
  MAX(CASE WHEN c.Sub_code='TCS1212' THEN c.Grade END) AS TCS1212,
  MAX(CASE WHEN c.Sub_code='TCS1233' THEN c.Grade END) AS TCS1233,
  COALESCE(s.SGPA,0.00) AS SGPA,
  COALESCE(s.SGPA,0.00) AS CGPA
FROM capped c
LEFT JOIN sgpa s ON s.Reg_no = c.Reg_no
GROUP BY c.Reg_no, s.SGPA
ORDER BY c.Reg_no;

        WHEN ca.ICT1222 < 20 THEN 'NE*'
